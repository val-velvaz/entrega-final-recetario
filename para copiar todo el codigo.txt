# --- COMANDO PARA CONSOLIDAR CÓDIGO PARA CLAUDE ---

$RutaRaiz = "C:\Universidad\estructuras-de-datos"
$ArchivoSalida = "$RutaRaiz\todo_el_codigo_final2.txt"

# 1. Asegurar que el directorio de salida exista
Set-Location $RutaRaiz

# 2. Limpiar el archivo de salida antes de empezar
Remove-Item -Path $ArchivoSalida -ErrorAction SilentlyContinue

# 3. Recorrer y concatenar archivos usando codificación UTF8
$ListaArchivos = Get-ChildItem -Recurse -Include *.cpp,*.hpp | Select-Object -ExpandProperty FullName

$ListaArchivos | ForEach-Object {
    $RutaActual = $_
    $Contenido = Get-Content $RutaActual -Raw
    
    # Escribir separador y nombre del archivo
    Add-Content -Path $ArchivoSalida -Value "===== $($RutaActual) ====="
    
    # Escribir el contenido (uso de -Encoding UTF8 para evitar problemas de caracteres)
    Add-Content -Path $ArchivoSalida -Value $Contenido -Encoding UTF8
    
    # Agregar un salto de línea
    Add-Content -Path $ArchivoSalida -Value "`n"
}

Write-Host "Archivo de contexto para Claude creado con éxito en: $ArchivoSalida" -ForegroundColor Green
